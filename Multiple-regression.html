Simple linear regression

# Sample size
n = 1000

# Continuous predictor
x1 = rnorm(n)

# Continuous outcome
y = 2 * x1 + rnorm(n)

# Simple linear regression model
slrm = lm(y ~ x1)

# Plot model fit
plot(x1, y)
abline(coef = coef(slrm), col = 2)

# Model summary
summary(slrm)
